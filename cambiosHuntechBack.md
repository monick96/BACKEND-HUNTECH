### create tables para postgres en supabase
```
create table public.desarrollador (
  id varchar(60),
  email varchar(100),
  nombre varchar(100),
  apellido varchar(100),
  descripcion text,
  fecha_nacimiento varchar(100),
  rol varchar(255),
  skills varchar(255),
  created_at timestamptz default now()
);

create table public.gerente (
  id_gerente varchar(60),
  id_proyecto varchar(60),
  nombre varchar(90),
  email varchar(100),
  descripcion text
);

create table public.institucion_educativa (
  id varchar(60),
  email varchar(100),
  nombre varchar(255)
);

create table public.proyecto (
  id varchar(60) primary key,
  nombre varchar(500),
  description text,
  info_link varchar(500),
  buscando_devs boolean not null default false,
  contratos text,
  id_gerente varchar(60),
  email_gerente varchar(100)
);

create table public.carrera (
  id bigint generated by default as identity primary key,
  nombre varchar(255),
  info_link varchar(500),
  id_institucion_educativa varchar(60),
  status varchar(50)
);

create table public.carrera_x_desarrollador (
  id_desarrollador varchar(60),
  id_carrera varchar(60),
  start_date varchar(100),
  end_date varchar(100),
  isvalidated boolean not null default false
);

create table public.contrato (
  id bigint generated by default as identity primary key,
  tipo varchar(100),
  titulo varchar(255),
  descripcion text,
  tiene_postulaciones boolean,
  postulaciones text,
  esta_ocupado boolean,
  pasante_email varchar(100),
  proyecto_id varchar(60),
  start_date varchar(100),
  end_date varchar(100)
);

VARCHAR(MAX) -> TEXT
BIT -> BOOLEAN
INT IDENTITY(1,1) -> SERIAL PRIMARY KEY
GETDATE() -> NOW()
```
